def fizzbuzz_plus(n):
    for i in range(1, n + 1):
        result = ""
        if i % 3 == 0:
            result += "Fizz"
        if i % 5 == 0:
            result += "Buzz" 
        if i % 7 == 0:
            result += "Boom"
        if not result:
            result = str(i)
        print(result)

# Test first 20 numbers
fizzbuzz_plus(20)
