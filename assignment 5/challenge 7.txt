def rotate_array(arr, k):
    n = len(arr)
    k = k % n  # Handle k > n
    
    # Reverse entire array
    arr[:] = arr[::-1]
    # Reverse first k elements
    arr[:k] = arr[:k][::-1]  
    # Reverse remaining n-k elements
    arr[k:] = arr[k:][::-1]
    
    return arr

# Test
arr = [1, 2, 3, 4, 5, 6, 7]
rotate_array(arr, 3)
print(arr)  # Output: [5, 6, 7, 1, 2, 3, 4]
