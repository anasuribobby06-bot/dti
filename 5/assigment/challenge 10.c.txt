#include <stdio.h>

/* Function to swap two numbers using pointers */
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int x = 10, y = 20;

    /* Print values and addresses before swap */
    printf("Before Swap:\n");
    printf("x = %d, Address of x = %p\n", x, (void*)&x);
    printf("y = %d, Address of y = %p\n\n", y, (void*)&y);

    /* Swap using pointers */
    swap(&x, &y);

    /* Print values and addresses after swap */
    printf("After Swap:\n");
    printf("x = %d, Address of x = %p\n", x, (void*)&x);
    printf("y = %d, Address of y = %p\n\n", y, (void*)&y);

    /* Array traversal using pointers (NO indexing) */
    int arr[] = {1, 2, 3, 4, 5};
    int *ptr = arr;
    int size = sizeof(arr) / sizeof(arr[0]);

    printf("Array Traversal Using Pointers:\n");
    for (int i = 0; i < size; i++) {
        printf("Value: %d, Address: %p\n", *ptr, (void*)ptr);
        ptr++;  // move pointer to next element
    }

    return 0;
}
