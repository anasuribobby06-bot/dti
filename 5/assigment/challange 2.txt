#include <stdio.h>  // Standard I/O library for scanf and printf

int main() {        // Program entry point
    char name[50];  // Array to store name string (max 49 chars + null terminator)
    int age;        // Integer variable for age
    float salary;   // Float variable for salary
    
    // Prompt and read inputs using correct format specifiers
    printf("Enter your name: ");
    scanf("%s", name);           // %s for string (no & needed for char arrays)
    
    printf("Enter your age: ");
    scanf("%d", &age);           // %d for integer, & passes address
    
    printf("Enter your salary: ");
    scanf("%f", &salary);        // %f for float, & passes address
    
    // Print back neatly formatted
    printf("\nName: %s\n", name);
    printf("Age: %d\n", age);
    printf("Salary: %.2f\n", salary);  // .2 limits to 2 decimal places
    
    return 0;       // Successful exit
}
